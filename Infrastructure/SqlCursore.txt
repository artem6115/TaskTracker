fetch MyCursor into @Id, @StatusTask, @UserId

while @@FETCH_STATUS <> 0
begin
	print('Hellow')
	If (@UserId is null)
		UPDATE Tasks SET StatusTask=0 WHERE PreviousTaskId = @IdChanged AND Id = @Id

end

end
